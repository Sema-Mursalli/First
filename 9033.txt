#include <bits/stdc++.h>       \\9633

using namespace std;

int main()
{
    int n,say=0;
    cin>>n;
    int a[n];
    vector<int>v;
    
    for(int i=0 ; i<n ; i++)
    {
        cin>>a[i];
        v.push_back(a[i]);
        v.push_back(a[i]-1);
        v.push_back(a[i]+1);
        
    }
    sort(v.begin(),v.end());
    int maxi=0;
    for(int i=0 ; i<v.size(); i++)
    {
        if(v[i]==v[i+1])
        say++;
        else
        say=0;
        if(say>maxi)
        maxi=say;
    }
    cout<<maxi+1;
    }
